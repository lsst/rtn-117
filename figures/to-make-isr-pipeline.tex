% #1 = node options, #2 = name, #3 = step number, #4 = label text (use \\ between lines)
\newcommand{\stepnode}[4][]{%
  \node[#1, minimum height=3.0em, label={[stepnum, xshift=4pt, yshift=-3pt]north west:#3}] (#2) {%
    \vspace{0.7em}\shortstack[c]{#4}%
  }%
}
